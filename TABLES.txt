//TABLE CREATION CODES (copy & paste from below)

CREATE TABLE "categories" ("categoryID" INTEGER PRIMARY KEY  NOT NULL  DEFAULT (null) ,"name" VARCHAR NOT NULL ,"img" VARCHAR)

CREATE TABLE "itemFunctions" ("functionID" INTEGER PRIMARY KEY  NOT NULL  UNIQUE , "name" VARCHAR NOT NULL  UNIQUE )

CREATE TABLE "items" ("itemID" INTEGER PRIMARY KEY  NOT NULL  DEFAULT (null) ,"name" VARCHAR NOT NULL ,"desc" TEXT,"img" VARCHAR,"rarityID" INTEGER NOT NULL  DEFAULT (null) ,"categoryID" INTEGER NOT NULL  DEFAULT (null) ,"functionID" INTEGER NOT NULL  DEFAULT (null) ,"maxStack" INTEGER NOT NULL ,"canTrade" BOOL NOT NULL )

CREATE TABLE "ranks" ("rankID" INTEGER PRIMARY KEY  NOT NULL  DEFAULT (null) ,"name" VARCHAR NOT NULL ,"color" VARCHAR NOT NULL )

CREATE TABLE "rarities" ("rarityID" INTEGER PRIMARY KEY  NOT NULL  DEFAULT (null) ,"name" VARCHAR NOT NULL ,"color" VARCHAR NOT NULL )

CREATE TABLE "shopCosts" ("itemID" INTEGER  NOT NULL , "costID" INTEGER NOT NULL , "quantity" INTEGER NOT NULL )

CREATE TABLE "shopStock" ("stockID" INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  UNIQUE , "itemID" INTEGER NOT NULL , "quantity" INTEGER, "balCost" INTEGER, "costsItems" BOOL NOT NULL )

CREATE TABLE "transactions" ("transactionID" INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  UNIQUE , "userID" INTEGER NOT NULL , "itemIDsBought" VARCHAR NOT NULL , "quantitiesBought" VARCHAR NOT NULL , "balPaid" INTEGER, "itemIDsPaid" VARCHAR, "quantitiesPaid" VARCHAR, "time" INTEGER NOT NULL )

CREATE TABLE "userInventories" ("userID" INTEGER  NOT NULL  DEFAULT (null) ,"itemID" INTEGER NOT NULL  DEFAULT (null) ,"quantity" INTEGER NOT NULL )

CREATE TABLE "users" ("userID" INTEGER PRIMARY KEY  NOT NULL  DEFAULT (null) ,"name" VARCHAR NOT NULL ,"pass" VARCHAR NOT NULL ,"isAdmin" BOOL NOT NULL ,"rankID" INTEGER NOT NULL  DEFAULT (null) ,"balance" INTEGER NOT NULL  DEFAULT (0) ,"salt" VARCHAR)
